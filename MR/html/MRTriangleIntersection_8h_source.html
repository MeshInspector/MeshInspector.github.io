43

45  return false; // triangle ABC is located at one side of the plane DEF
46

49  else if( abc_fd )

122 std::optional<TriIntersectResult> rayTriangleIntersect_( const Vector3<T>& oriA, const Vector3<T>& oriB, const Vector3<T>& oriC,
123  const IntersectionPrecomputes<T>& prec )
124 {
125  const T& Sx = prec.Sx;
126  const T& Sy = prec.Sy;
127  const T& Sz = prec.Sz;
128 
129  const T Ax = oriA[prec.idxX] - Sx * oriA[prec.maxDimIdxZ];
130  const T Ay = oriA[prec.idxY] - Sy * oriA[prec.maxDimIdxZ];
131  const T Bx = oriB[prec.idxX] - Sx * oriB[prec.maxDimIdxZ];
132  const T By = oriB[prec.idxY] - Sy * oriB[prec.maxDimIdxZ];
133  const T Cx = oriC[prec.idxX] - Sx * oriC[prec.maxDimIdxZ];
134  const T Cy = oriC[prec.idxY] - Sy * oriC[prec.maxDimIdxZ];
135 
136  T U = Cx * By - Cy * Bx;
137  T V = Ax * Cy - Ay * Cx;
138  T W = Bx * Ay - By * Ax;
139 
140  if( U < T( 0 ) || V < T( 0 ) || W < T( 0 ) )
141  {
142  if( U > T( 0 ) || V > T( 0 ) || W > T( 0 ) )
143  {
144  return std::nullopt;
145  }
146  }
147 
148  T det = U + V + W;
149  if( det == T( 0 ) )
150  return std::nullopt;
151  const T Az = Sz * oriA[prec.maxDimIdxZ];
152  const T Bz = Sz * oriB[prec.maxDimIdxZ];
153  const T Cz = Sz * oriC[prec.maxDimIdxZ];
154  const T t = U * Az + V * Bz + W * Cz;
155 
156  auto invDet = T( 1 ) / det;
157  return TriIntersectResult( float( U * invDet ), float( V * invDet ), float( t * invDet ) );
158 }
