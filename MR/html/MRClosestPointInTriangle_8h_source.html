24 static std::pair<Vector3f, TriPointf> closestPointInTriangle( const Vector3f& p, const Vector3f& a, const Vector3f& b, const Vector3f& c )
25 {
26  const Vector3f ab = b - a;
27  const Vector3f ac = c - a;
28  const Vector3f ap = p - a;
29 
30  const float d1 = dot( ab, ap );
31  const float d2 = dot( ac, ap );
32  if ( d1 <= 0.f && d2 <= 0.f )
33  return { a, { 0, 0 } };
34 
35  const Vector3f bp = p - b;
36  const float d3 = dot( ab, bp );
37  const float d4 = dot( ac, bp );
38  if ( d3 >= 0.f && d4 <= d3 )
39  return { b, { 1, 0 } };
40 
41  const Vector3f cp = p - c;
42  const float d5 = dot( ab, cp );
43  const float d6 = dot( ac, cp );
44  if ( d6 >= 0.f && d5 <= d6 )
45  return { c, { 0, 1 } };
46

48  if ( vc <= 0.f && d1 >= 0.f && d3 <= 0.f )
49  {
50  const float v = d1 / ( d1 - d3 );
51  return { a + v * ab, { v, 0 } };
52  }
53 
54  const float vb = d5 * d2 - d1 * d6;
55  if ( vb <= 0.f && d2 >= 0.f && d6 <= 0.f )
56  {
57  const float v = d2 / ( d2 - d6 );
58  return { a + v * ac, { 0, v } };
59  }
60

63  {

66  }
