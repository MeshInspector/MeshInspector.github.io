202  struct PointInAllSpaces
203  {
204  // Screen space: window x[0, window.width] y[0, window.height]. [0,0] top-left corner of the window
205  // Z [0,1] - 0 is Dnear, 1 is Dfar
206  Vector3f screenSpace;
207 
208  // Viewport space: viewport x[0, viewport.width] y[0, viewport.height]. [0,0] top-left corner of the viewport with that Id
209  // Z [0,1] - 0 is Dnear, 1 is Dfar
210  Vector3f viewportSpace;
211  ViewportId viewportId;
212 
213  // Clip space: viewport xyz[-1.f, 1.f]. [0, 0, -1] middle point of the viewport on Dnear. [0, 0, 1] middle point of the viewport on Dfar.
214  // [-1, -1, -1] is lower left Dnear corner; [1, 1, 1] is upper right Dfar corner
215  Vector3f clipSpace;
216 
217  // Camera space: applied view affine transform to world points xyz[-inf, inf]. [0, 0, 0] middle point of the viewport on Dnear.
218  // X axis goes on the right. Y axis goes up. Z axis goes backward.
219  Vector3f cameraSpace;
220 
221  // World space: applied model transform to Mesh(Point Cloud) vertices xyz[-inf, inf].
222  Vector3f worldSpace;
223 
224  // Model space: coordinates as they stored in the model of VisualObject
225  std::shared_ptr<VisualObject> obj;
226  PointOnFace pof;
227  };
