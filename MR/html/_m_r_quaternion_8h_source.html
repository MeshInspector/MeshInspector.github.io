81 constexpr Quaternion<T>::Quaternion( const Matrix3<T> & m )
82 {
83  // https://www.euclideanspace.com/maths/geometry/rotations/conversions/matrixToQuaternion/
84  const auto tr = m.trace();
85 
86  if (tr > 0)
87  {
88  auto S = std::sqrt( tr + 1 ) * 2;
89  a = T(0.25) * S;
90  b = ( m.z.y - m.y.z ) / S;
91  c = ( m.x.z - m.z.x ) / S;
92  d = ( m.y.x - m.x.y ) / S;
93  }
94  else if ( m.x.x > m.y.y && m.x.x > m.z.z )
95  {
96  auto S = std::sqrt( 1 + m.x.x - m.y.y - m.z.z ) * 2;
97  a = ( m.z.y - m.y.z ) / S;
98  b = T(0.25) * S;
99  c = ( m.x.y + m.y.x ) / S;
100  d = ( m.x.z + m.z.x ) / S;
101  }
102  else if (m.y.y > m.z.z)
103  {
104  auto S = std::sqrt( 1 + m.y.y - m.x.x - m.z.z ) * 2;
105  a = ( m.x.z - m.z.x ) / S;
106  b = ( m.x.y + m.y.x ) / S;
107  c = T(0.25) * S;
108  d = ( m.y.z + m.z.y ) / S;
109  }
110  else
111  {
112  auto S = std::sqrt( 1 + m.z.z - m.x.x - m.y.y ) * 2;
113  a = ( m.y.x - m.x.y ) / S;
114  b = ( m.x.z + m.z.x ) / S;

120 template <typename T>
121 constexpr Quaternion<T>::Quaternion( const Vector3<T>& from, const Vector3<T>& to) noexcept
122 {
123  //https://stackoverflow.com/questions/1171849/finding-quaternion-representing-the-rotation-from-one-vector-to-another
124  a = dot( from, to );
125  auto cr = cross( from, to );
126  if( cr.x == 0 && cr.y == 0 && cr.z == 0 )
127  {
128  if( a >= 0 )
129  {
130  // parallel co-directional vectors
131  // zero rotation
132  a = 1; b = 0; c = 0; d = 0;
133  }
134  else
135  {
136  // parallel opposing vectors
137  // any perpendicular axis Pi rotation
138  auto perp = cross( from, from.furthestBasisVector() );
139  a = 0; b = perp.x; c = perp.y; d = perp.z;
140  normalize();
141  }
142  }
143  else
144  {
145  a += std::sqrt( from.lengthSq() * to.lengthSq() );
146  b = cr.x; c = cr.y; d = cr.z;
147  normalize();
148  }
149 }

151 template <typename T>
152 Quaternion<T> Quaternion<T>::slerp( Quaternion q0, Quaternion q1, T t )
153 {
154  // https://en.wikipedia.org/wiki/Slerp
155  q0.normalize();
156  q1.normalize();
157 
158  T cosTheta = std::clamp( dot( q0, q1 ), T(-1), T(1) );
159  if ( cosTheta < 0 )
160  {
161  q0 = -q0;
162  cosTheta = -cosTheta;
163  }
164  T theta = std::acos( cosTheta );
165  T sinTheta = std::sin( theta );
166  if ( sinTheta <= 0 )
167  return lerp( q0, q1, t ).normalized();
168 
169  return std::sin( theta * ( 1 - t ) ) / sinTheta * q0 + std::sin( theta * t ) / sinTheta * q1;
170 };
