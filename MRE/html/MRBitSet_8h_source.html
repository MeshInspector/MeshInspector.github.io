28  // this accessor automatically adjusts the size of the set to include i-th element
29  void autoResizeSet( size_t pos, bool val = true )
30  {
31  auto sz = size();
32  if ( pos == sz )
33  {
34  push_back( val );
35  return;
36  }
37  if ( pos > sz )
38  {
39  if ( capacity() <= pos && sz > 0 )
40  {
41  while ( pos < sz )
42  sz <<= 1;
43  reserve( sz );
44  }
45  resize( pos + 1 );
46  }
47  set( pos, val );
48  }
49 
50  // same as autoResizeSet(...) and returns previous value of pos-bit
51  bool autoResizeTestSet( size_t pos, bool val = true )
52  {
53  bool const b = test( pos );
54  if ( b != val )
55  autoResizeSet( pos, val );
56  return b;
57  }
