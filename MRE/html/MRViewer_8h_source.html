224  struct PointInAllSpaces

226  // Screen space: window x[0, window.width] y[0, window.height]. [0,0] top-left corner of the window
227  // Z [0,1] - 0 is Dnear, 1 is Dfar

229

232  Vector3f viewportSpace;

236  // [-1, -1, -1] is lower left Dnear corner; [1, 1, 1] is upper right Dfar corner

343  struct SignalStopHandler
344  {
345  using result_type = bool;
346 
347  template<typename Iter>
348  bool operator()( Iter first, Iter last ) const
349  {
350  while ( first != last )
351  {
352  if ( *first )
353  return true; // slots execution stops if one returns true
354  ++first;
355  }
356  return false;
357  }
358  };
