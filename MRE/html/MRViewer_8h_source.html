425 
426  template<typename Iter>
427  bool operator()( Iter first, Iter last ) const
428  {
429  while ( first != last )
430  {
431  if ( *first )
432  return true; // slots execution stops if one returns true
433  ++first;
434  }
435  return false;
436  }
