210 template<typename T>
211 struct RibbonMenuItemCall
212 {
213  template<typename Func>
214  RibbonMenuItemCall( Func f )
215  {
216  std::shared_ptr<RibbonMenu> menu = std::dynamic_pointer_cast< RibbonMenu >( Viewer::instanceRef().getMenuPlugin() );
217  if ( menu )
218  {
219  static_assert( std::is_base_of_v<RibbonMenuItem, T> );
220  const auto& items = menu->getRibbonSchema().items;
221  for ( const auto& item : items )
222  {
223  auto plugin = std::dynamic_pointer_cast<T>( item.second.item );
224  if ( !plugin )
225  continue;
226  f( plugin );
227  }
228  }
229  }
