283// Calculates points for a custom curve, from a function you provide. You can then draw this curve as a `line()`.
284// The points are appended into `pointBuffer` (and for convenience the new points are also returned as `.midPoints`;
285// the first and the last point are not inserted into the vector and are not included in `.midPoints`, they sit in `.a` and `.b`).
286// You should reuse `pointBuffer` between calls for better performance (and `.clear()` it each time you finish drawing the resulting curve,
287// which conveniently preserves vector capacity). Or, if you don't care, just pass an empty vector, and keep it alive as long as you need the curve.
288// `stateA` and `stateB` can have any type, they describe the beginning and the end of the curve respectively. They might often be `0.f` and `1.f`.
289// Let `S` be the type of `stateA`/`stateB` (more on that belw).
290// `stateToPoint` is a lambda `(const S &s) -> ImVec2`. It converts a state into a curve point coordinates.
291// `bisectState` is a lambda `(const S &a, const S &b, int depth) -> S`. Given two states, it produces the state between them. `depth` starts at `0`.
292// `onInsertPoint`, if specified, is a lambda `(ImVec2 point, const S &state) -> void`. It's called right before a point is added into the list.
293// It's actually possible to have multiple state types instead of a single `S`, and template your functions (e.g. to track bisection depth in the type system, etc).
294// Example usage:
295// prepareCurve( params, buf, 0, PI*2, [](float x){return std::sin(x);}, [](float a, float b, int /*depth*/){return (a+b)/2;} )
296template <typename A, typename B, typename F, typename G, typename H = std::nullptr_t>
297[[nodiscard]] PreparedCurve prepareCurve( const CurveParams& curveParams, std::vector<ImVec2>& pointBuffer, const A& stateA, const B& stateB,
298 F&& stateToPoint, G&& bisectState, H&& onInsertPoint = nullptr
299)
300{
301 const float pixelStepSq = curveParams.subdivisionStepPixels * curveParams.subdivisionStepPixels;
302
303 std::size_t firstIndex = pointBuffer.size();
304
305 auto makeCurve = [&]( auto makeCurve, int depth, const auto& stateA, const auto& stateB, ImVec2 pointA, ImVec2 pointB ) -> void
306 {
307 // Check if we need to subdivide.
308 if ( depth < curveParams.maxSubdivisionDepth && ( depth < curveParams.minSubdivisionDepth || ImGuiMath::lengthSq( pointB - pointA ) > pixelStepSq ) )
309 {
310 // Do subdivide.
311
312 auto midState = bisectState( stateA, stateB, int( depth ) ); // A cast to prevent modification.
313 ImVec2 midPoint = stateToPoint( midState );
314
315 makeCurve( makeCurve, depth + 1, stateA, midState, pointA, midPoint );
316 makeCurve( makeCurve, depth + 1, midState, stateB, midPoint, pointB );
317 }
318 else
319 {
320 // No subdivide.
321
322 onInsertPoint( pointB, stateB );
323 pointBuffer.push_back( pointB );
324 }
325
326 };
327 ImVec2 firstPoint = stateToPoint( stateA );
328 makeCurve( makeCurve, 0, stateA, stateB, firstPoint, stateToPoint( stateB ) );
329
330 PreparedCurve ret{ .a = firstPoint, .b = pointBuffer.back() };
331 pointBuffer.pop_back();
332 ret.midPoints = { pointBuffer.data() + firstIndex, pointBuffer.data() + pointBuffer.size() };
333 return ret;
334}
