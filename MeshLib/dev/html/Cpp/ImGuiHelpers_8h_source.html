50inline bool Combo( const char* label, int* idx, std::function<const char*( int )> getter, int itemsCount )
51{
52 auto func = [] ( void* data, int i ) -> const char*
53 {
54 auto& getter = *reinterpret_cast< std::function<const char*( int )> * >( data );
55 const char* s = getter( i );
56 if ( s )
57 return s;
58 else
59 {
60 assert( false && "Combo: getter return nullptr" );
61 return "";
62 }
63 };
64 return Combo( label, idx, func, reinterpret_cast< void* >( &getter ), itemsCount );
