33static auto vector_getter = [] ( void* vec, int idx ) -> const char*
34{
35 auto& vector = *static_cast< std::vector<std::string>* >( vec );
36 if ( idx < 0 || idx >= static_cast< int >( vector.size() ) )
37 {
38 assert( false && "Combo: vector_getter invalid index" );
39 return "";
40 }
41 return vector.at( idx ).c_str();
42};
43
44inline bool Combo(const char* label, int* idx, const std::vector<std::string>& values)
45{
46 if (values.empty()) { return false; }
47 return Combo(label, idx, vector_getter,
48 const_cast<void *>( static_cast<const void*>(&values) ), (int)values.size());
49}
50
51inline bool Combo(const char* label, int* idx, std::function<const char *(int)> getter, int items_count)
52{
53 auto func = [](void* data, int i) -> const char*
54 {
55 auto &getter = *reinterpret_cast<std::function<const char *(int)> *>(data);
56 const char *s = getter(i);
57 if ( s )
58 return s;
59 else
60 {
61 assert( false && "Combo: getter return nullptr" );
62 return "";
63 }
64 };
65 return Combo(label, idx, func, reinterpret_cast<void *>(&getter), items_count);
66}
