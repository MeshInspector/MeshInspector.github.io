238template <typename T>
239bool checkboxFlags( const char* label, T& target, T flags )
240{
241 bool value = bool( target & flags );
242 bool mixed = value && ( target & flags ) != flags;
243 if ( checkboxMixed( label, &value, mixed ) )
244 {
245 if ( value )
246 target |= flags;
247 else
248 target &= ~flags;
249 return true;
250 }
251 return false;
