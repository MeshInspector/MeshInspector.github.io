211template <typename T>
212bool checkboxFlags( const char* label, T& target, T flags )
213{
214 bool value = bool( target & flags );
215 bool mixed = value && ( target & flags ) != flags;
216 if ( checkboxMixed( label, &value, mixed ) )
217 {
218 if ( value )
219 target |= flags;
220 else
221 target &= ~flags;
222 return true;
223 }
224 return false;

320 // Whether `T` is a scalar type that we can use with our widgets.
321 template <typename T>
322 concept Scalar = std::is_arithmetic_v<T> && !std::is_same_v<T, bool>;
323
324 // Whether `T` is a scalar or vector that we can use with our widgets.
325 template <typename T>
326 concept VectorOrScalar = Scalar<typename VectorTraits<T>::BaseType>;
327
328 // Whether `Bound` is a valid min/max bound for `Target`.
329 // That is, either the same type, or if `Target` is a vector, `Bound` can also be a scalar of the same type.
330 template <typename Bound, typename Target>
331 concept ValidBoundForTargetType =
332 std::same_as<Bound, Target> ||
333 ( VectorTraits<Bound>::size == 1 && std::same_as<typename VectorTraits<Bound>::BaseType, typename VectorTraits<Target>::BaseType> );
334
335 // Whether `Speed` is a valid drag speed type for `Target`.
336 // That is, either a single/vector of `float` or the same type as target (or its element if it's a vector).
337 template <typename Speed, typename Target>
338 concept ValidDragSpeedForTargetType =
339 std::same_as<Speed, typename VectorTraits<Target>::BaseType> || std::same_as<Speed, float> ||
340 std::same_as<Speed, Target> || std::same_as<Speed, typename VectorTraits<Target>::template ChangeBase<float>>;

342 // A common code for sliders and other widgets dealing with measurement units.
343 // `E` must be explicitly set to a measurement unit enum. The other template parameters are deduced.
344 // `label` is the widget label, `v` is the target value.
345 // `func` draws the widget for an individual scalar. We call it more than once for vectors.
346 // `func` is `( const char* label, auto& elem, int i ) -> bool`.
347 // It receives `elem` already converted to the display units (so you must convert min/max bounds manually). `i` is the element index for vectors.
348 // When `v` is integral, `func` will be instantiated for both integral and floating-point element type. The latter is required if we're doing conversions.
349 // NOTE: For integral `v`, in `func` you must look at the type of `elem` and convert your min/max bounds (etc) to the same type.
350 // Notice `unitParams` being accepted by an lvalue reference. For convenience, we reset the `sourceUnit` in it before calling the user callback,
351 // since at that point no further conversions are necessary.
352 template <UnitEnum E, VectorOrScalar T, typename F>
353 [[nodiscard]] bool unitWidget( const char* label, T& v, UnitToStringParams<E>& unitParams, F&& func );
