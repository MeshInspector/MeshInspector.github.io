160// return real GL value for DepthFunction
161// default is less
162inline int getDepthFunctionLess( DepthFunction funcType )
163{
164 switch ( funcType )
165 {
166 case DepthFunction::Default:
167 case DepthFunction::Less:
168 return GL_LESS;
169 case DepthFunction::Never:
170 return GL_NEVER;
171 case DepthFunction::LessOrEqual:
172 return GL_LEQUAL;
173 case DepthFunction::Equal:
174 return GL_EQUAL;
175 case DepthFunction::GreaterOrEqual:
176 return GL_GEQUAL;
177 case DepthFunction::Greater:
178 return GL_GREATER;
179 case DepthFunction::Always:
180 return GL_ALWAYS;
181 case DepthFunction::NotEqual:
182 return GL_NOTEQUAL;
183 default:
184 return 0;
185 }
186}
187
188// return real GL value for DepthFunction
189// default is less or equal
190inline int getDepthFunctionLEqual( DepthFunction funcType )
191{
192 if ( funcType == DepthFunction::Default )
193 return GL_LEQUAL;
194 return getDepthFunctionLess( funcType );
195}
