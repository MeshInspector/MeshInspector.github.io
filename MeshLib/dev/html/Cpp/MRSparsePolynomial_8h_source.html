143{
144 std::map<D,C> resMap;
145 for ( const auto & [degA, cfA] : a.map_ )
146 {
147 assert( cfA != 0 );

150 assert( cfB != 0 );

152 if ( deg > M )
153 break;

156 if ( !inserted )
157 {

159 if ( sum != 0 )
160 it->second = sum;
161 else
162 resMap.erase( it );
163 }
