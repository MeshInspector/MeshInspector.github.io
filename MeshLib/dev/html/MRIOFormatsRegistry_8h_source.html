45template <typename Processor>
46class FormatRegistry
47{
48public:
49 // get all registered filters
50 static const IOFilters& getFilters()
51 {
52 return get_().filters_;
53 }
54
55 // get a registered loader for the filter
56 static Processor getProcessor( const IOFilter& filter )
57 {
58 const auto& processors = get_().processors_;
59 auto it = processors.find( filter );
60 if ( it != processors.end() )
61 return it->second;
62 else
63 return {};
64 }
65
66 // get a registered loader for the extension
67 static Processor getProcessor( const std::string& extension )
68 {
69 const auto& processors = get_().processors_;
70 // TODO: extension cache
71 auto it = std::find_if( processors.begin(), processors.end(), [&extension] ( auto&& item )
72 {
73 const auto& [filter, _] = item;
74 return filter.extensions.find( extension ) != std::string::npos;
75 } );
76 if ( it != processors.end() )
77 return it->second;
78 else
79 return {};
80 }
81
82 // register or update a loader for the filter
83 static void setProcessor( const IOFilter& filter, Processor processor, int8_t priorityScore = 0 )
84 {
85 auto& processors = get_().processors_;
86 auto it = processors.find( filter );
87 if ( it != processors.end() )
88 {
89 it->second = processor;
90 }
91 else
92 {
93 processors.emplace( filter, processor );
94
95 auto& filters = get_().filterPriorityQueue_;
96 filters.emplace( priorityScore, filter );
97 get_().updateFilterList_();
98 }
99 }
100
101private:
102 FormatRegistry() = default;
103 ~FormatRegistry() = default;
104
105 static FormatRegistry<Processor>& get_()
106 {
107 static FormatRegistry<Processor> instance;
108 return instance;
109 }
110
111 void updateFilterList_()
112 {
113 filters_.clear();
114 filters_.reserve( filterPriorityQueue_.size() );
115 for ( const auto& [_, filter] : filterPriorityQueue_ )
116 filters_.emplace_back( filter );
117 }
118
119 std::map<IOFilter, Processor> processors_;
120 std::multimap<int8_t, IOFilter> filterPriorityQueue_;
121 std::vector<IOFilter> filters_;
122};
