115template <typename V>
116class VoxelsVolumeCachingAccessor
117{
118public:
119 using VolumeType = V;
120 using ValueType = typename V::ValueType;
121
122 struct Parameters
123 {
125 size_t preloadedLayerCount = 1;
126 };
127
128 VoxelsVolumeCachingAccessor( const VoxelsVolumeAccessor<V>& accessor, const VolumeIndexer& indexer, Parameters parameters = {} )
129 : accessor_( accessor )
130 , indexer_( indexer )
131 , params_( std::move( parameters ) )
132 , layers_( params_.preloadedLayerCount )
133 , firstLayerVoxelId_( params_.preloadedLayerCount )
134 {
135 assert( params_.preloadedLayerCount > 0 );
136 for ( auto & l : layers_ )
137 l.resize( indexer_.sizeXY() );
138 }
139
141 [[nodiscard]] int currentLayer() const
142 {
143 return z_;
144 }
145
147 void preloadLayer( int z )
148 {
149 assert( 0 <= z && z < indexer_.dims().z );
150 z_ = z;
151 for ( auto layerIndex = 0; layerIndex < layers_.size(); ++layerIndex )
152 {
153 if ( indexer_.dims().z <= z_ + layerIndex )
154 break;
155 preloadLayer_( layerIndex );
156 }
157 }
158
160 void preloadNextLayer()
161 {
162 z_ += 1;
163 for ( auto i = 0; i + 1 < layers_.size(); ++i )
164 {
165 std::swap( layers_[i], layers_[i + 1] );
166 firstLayerVoxelId_[i] = firstLayerVoxelId_[i + 1];
167 }
168 if ( z_ + params_.preloadedLayerCount - 1 < indexer_.dims().z )
169 preloadLayer_( params_.preloadedLayerCount - 1 );
170 }
171
173 ValueType get( const VoxelLocation & loc ) const
174 {
175 const auto layerIndex = loc.pos.z - z_;
176 assert( 0 <= layerIndex && layerIndex < layers_.size() );
177 assert( loc.id >= firstLayerVoxelId_[layerIndex] );
178 assert( loc.id < firstLayerVoxelId_[layerIndex] + indexer_.sizeXY() );
179 return layers_[layerIndex][loc.id - firstLayerVoxelId_[layerIndex]];
180 }
181
182private:
183 [[nodiscard]] size_t toLayerIndex( const Vector3i& pos ) const
184 {
185 return indexer_.toVoxelId( { pos.x, pos.y, 0 } );
186 }
187
188 void preloadLayer_( size_t layerIndex )
189 {
190 assert( layerIndex < layers_.size() );
191 auto& layer = layers_[layerIndex];
192 const auto z = z_ + (int)layerIndex;
193 const auto& dims = indexer_.dims();
194 assert( 0 <= z && z < dims.z );
195 auto loc = indexer_.toLoc( Vector3i{ 0, 0, z } );
196 firstLayerVoxelId_[layerIndex] = loc.id;
197 size_t n = 0;
198 for ( loc.pos.y = 0; loc.pos.y < dims.y; ++loc.pos.y )
199 for ( loc.pos.x = 0; loc.pos.x < dims.x; ++loc.pos.x, ++loc.id, ++n )
200 layer[n] = accessor_.get( loc );
201 }
202
203private:
204 const VoxelsVolumeAccessor<V>& accessor_;
205 VolumeIndexer indexer_;
206 Parameters params_;
207
208 int z_ = -1;
209 std::vector<std::vector<ValueType>> layers_;
210 std::vector<VoxelId> firstLayerVoxelId_;
211};
