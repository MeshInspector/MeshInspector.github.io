256// return real GL value for DepthFunction
257// default is less
258inline int getDepthFunctionLess( DepthFunction funcType )
259{
260 switch ( funcType )
261 {
262 case DepthFunction::Default:
263 case DepthFunction::Less:
264 return GL_LESS;
265 case DepthFunction::Never:
266 return GL_NEVER;
267 case DepthFunction::LessOrEqual:
268 return GL_LEQUAL;
269 case DepthFunction::Equal:
270 return GL_EQUAL;
271 case DepthFunction::GreaterOrEqual:
272 return GL_GEQUAL;
273 case DepthFunction::Greater:
274 return GL_GREATER;
275 case DepthFunction::Always:
276 return GL_ALWAYS;
277 case DepthFunction::NotEqual:
278 return GL_NOTEQUAL;
279 default:
280 return 0;
281 }
282}
283
284// return real GL value for DepthFunction
285// default is less or equal
286inline int getDepthFunctionLEqual( DepthFunction funcType )
287{
288 if ( funcType == DepthFunction::Default )
289 return GL_LEQUAL;
290 return getDepthFunctionLess( funcType );
291}
