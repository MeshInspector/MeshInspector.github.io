58 // given t in [0,1] and rigid transformations, interpolates them spherically and produces another rigid transformation;
59 // p is the only point that will have straight line movement during interpolation
60 [[nodiscard]] static AffineXf3<T> slerp( const AffineXf3<T> & xf0, const AffineXf3<T> & xf1, T t, const Vector3<T> & p = {} )
61 {
62 auto xfA = slerp( xf0.A, xf1.A, t );
63 return { xfA, ( 1 - t ) * xf0( p ) + t * xf1( p ) - xfA * p };
64 }

67 Quaternion & operator /=( T s ) { return *this *= ( 1 / s ); }
68};

70template <typename T>
71constexpr Quaternion<T>::Quaternion( const Vector3<T> & axis, T angle ) noexcept
72{
73 a = std::cos( angle / 2 );
74 Vector3<T> im = std::sin( angle / 2 ) * axis.normalized();
75 b = im.x;
76 c = im.y;
77 d = im.z;
78}

80template <typename T>
81constexpr Quaternion<T>::Quaternion( const Matrix3<T> & m )
82{
83 // https://www.euclideanspace.com/maths/geometry/rotations/conversions/matrixToQuaternion/
84 a = T(0.5) * std::sqrt( std::max( T(0), 1 + m.x.x + m.y.y + m.z.z ) );
85 b = T(0.5) * std::copysign( std::sqrt( std::max( T(0), 1 + m.x.x - m.y.y - m.z.z ) ), m.z.y - m.y.z );
86 c = T(0.5) * std::copysign( std::sqrt( std::max( T(0), 1 - m.x.x + m.y.y - m.z.z ) ), m.x.z - m.z.x );
87 d = T(0.5) * std::copysign( std::sqrt( std::max( T(0), 1 - m.x.x - m.y.y + m.z.z ) ), m.y.x - m.x.y );
88}

90template <typename T>
91constexpr Quaternion<T>::Quaternion( const Vector3<T>& from, const Vector3<T>& to) noexcept
92{
93 //https://stackoverflow.com/questions/1171849/finding-quaternion-representing-the-rotation-from-one-vector-to-another
94 a = dot( from, to );
95 auto cr = cross( from, to );
96 if( cr.x == 0 && cr.y == 0 && cr.z == 0 )
97 {
98 if( a >= 0 )
99 {
100 // parallel co-directional vectors
101 // zero rotation
102 a = 1; b = 0; c = 0; d = 0;
103 }
104 else
105 {
106 // parallel opposing vectors
107 // any perpendicular axis Pi rotation
108 auto perp = cross( from, from.furthestBasisVector() );
109 a = 0; b = perp.x; c = perp.y; d = perp.z;
110 normalize();
111 }
112 }
113 else
114 {
115 a += std::sqrt( from.lengthSq() * to.lengthSq() );
116 b = cr.x; c = cr.y; d = cr.z;
117 normalize();
118 }
119}

121template <typename T>
122Quaternion<T> Quaternion<T>::slerp( Quaternion q0, Quaternion q1, T t )
123{
124 // https://en.wikipedia.org/wiki/Slerp
125 q0.normalize();
126 q1.normalize();
127
128 T cosTheta = std::clamp( dot( q0, q1 ), T(-1), T(1) );
129 if ( cosTheta < 0 )
130 {
131 q0 = -q0;
132 cosTheta = -cosTheta;
133 }
134 T theta = std::acos( cosTheta );
135 T sinTheta = std::sin( theta );
136 if ( sinTheta <= 0 )
137 return lerp( q0, q1, t ).normalized();
138
139 return std::sin( theta * ( 1 - t ) ) / sinTheta * q0 + std::sin( theta * t ) / sinTheta * q1;
140}

142template <typename T>
143Quaternion<T>::operator Matrix3<T>() const
144{
145 Matrix3<T> res;
146 res.x = Vector3<T>{ a * a + b * b - c * c - d * d, 2*(b * c - a * d), 2*(b * d + a * c) };
147 res.y = Vector3<T>{ 2*(b * c + a * d), a * a + c * c - b * b - d * d, 2*(c * d - a * b) };
148 res.z = Vector3<T>{ 2*(b * d - a * c), 2*(c * d + a * b), a * a + d * d - b * b - c * c };
149 return res;
150}

257// returns closest to base canonical quaternion
258template<typename T>
259[[nodiscard]] Quaternion<T> getClosestCanonicalQuaternion( const Quaternion<T>& base ) noexcept
260{
261 Quaternion<T> baseInverse = base.normalized().inverse();
262 int closestIndex{0};
263 T maxCos = T(-2);
264 const Quaternion<T>* canonQuats = getCanonicalQuaternions<T>();
265 for ( int i = 0; i < 24; ++i )
266 {
267 const Quaternion<T>& canonQuat = canonQuats[i];
268 Quaternion<T> relativeQuat = canonQuat * baseInverse;
269 relativeQuat.normalize();
270 T cos = std::abs( relativeQuat.a );
271 if ( cos > maxCos )
272 {
273 maxCos = cos;
274 closestIndex = i;
275 }
276 }
277 return canonQuats[closestIndex];
278}

310// center point will be transformed to same point by both input and output transformations

313{

315 res.A = orthonormalized( xf.A );
316 res.b = xf( center ) - res.A * center;
