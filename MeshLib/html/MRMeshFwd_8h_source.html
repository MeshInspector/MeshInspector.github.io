3// Not zero _ITERATOR_DEBUG_LEVEL in Microsoft STL greatly reduce the performance of STL containers.
4// So we change its value to zero by default. A huge restriction with this is that
5// all other linked DLL's and LIBS' also need to define this symbol to remove STL debugging, see
6// 1) vcpkg/triplets/x64-windows-meshrus.cmake and
7// 2) MeshLib/source/common.props
8// If you would like not-zero _ITERATOR_DEBUG_LEVEL and
9// you know what you are doing (up to 100x slowdown),
10// please define MR_ITERATOR_DEBUG_LEVEL as well
11#if defined _MSC_VER
12 #if !defined _ITERATOR_DEBUG_LEVEL
13 #define _ITERATOR_DEBUG_LEVEL 0
14 #endif

16 #define MR_ITERATOR_DEBUG_LEVEL 0
17 #endif
