149template<typename T, typename F1, typename F2>
150void PolylineTopology::buildFromContours( const std::vector<std::vector<T>> & contours, F1 && reservePoints, F2 && addPoint )
151{
152 *this = {};
153 size_t size = 0;
154 std::vector<bool> closed;
155 closed.reserve( contours.size() );
156 int numNotEmpty = 0;
157 int numClosed = 0;
158 for ( const auto& c : contours )
159 {
160 if ( c.size() > 2 )
161 {
162 ++numNotEmpty;
163 closed.push_back( c.front() == c.back() );
164 }
165 else
166 {
167 closed.push_back( false );
168 }
169 size += c.size();
170 if ( closed.back() )
171 ++numClosed;
172 }
173
174 reservePoints( size - numClosed );
175 vertResize( size - numClosed );
176
177 for ( int i = 0; i < contours.size(); ++i )
178 {
179 const auto& c = contours[i];
180 if ( c.empty() )
181 continue;
182 const auto e0 = makeEdge();
183 const auto v0 = addPoint( c[0] );
184 setOrg( e0, v0 );
185 auto e = e0;
186 for ( int j = 1; j + 1 < c.size(); ++j )
187 {
188 const auto ej = makeEdge();
189 splice( ej, e.sym() );
190 const auto vj = addPoint( c[j] );
191 setOrg( ej, vj );
192 e = ej;
193 }
194 if ( closed[i] )
195 {
196 splice( e0, e.sym() );
197 }
198 else
199 {
200 const auto vx = addPoint( c.back() );
201 setOrg( e.sym(), vx );
202 }
203 }
204 assert( isConsistentlyOriented() );
205}

207template<typename T, typename F>
208std::vector<std::vector<T>> PolylineTopology::convertToContours( F && getPoint ) const
209{
210 std::vector<std::vector<T>> res;
211
212 UndirectedEdgeBitSet linesUsed;
213 linesUsed.autoResizeSet( UndirectedEdgeId{ undirectedEdgeSize() } );
214 linesUsed.flip();
215 for ( EdgeId e0 : linesUsed )
216 {
217 if ( isLoneEdge( e0 ) )
218 continue;
219
220 EdgeId curLine = e0;
221 while ( curLine != next( curLine ) ) // until contour start is reached
222 {
223 curLine = next( curLine ).sym();
224 if ( curLine == e0 )
225 break; // the contour is closed
226 }
227
228 linesUsed.set( curLine.undirected(), false );
229
230 EdgeId e = curLine;
231 std::vector<T> cont;
232 cont.push_back( getPoint( org( e ) ) );
233 for ( ;; )
234 {
235 e = e.sym();
236 cont.push_back( getPoint( org( e ) ) );
237 e = next( e );
238 if ( !linesUsed.test_set( e.undirected(), false ) )
239 break;
240 }
241 res.push_back( std::move( cont ) );
242 }
243
244 return res;
245}
