157template<typename T, typename F1, typename F2>
158void PolylineTopology::buildFromContours( const std::vector<std::vector<T>> & contours, F1 && reservePoints, F2 && addPoint )
159{
160 *this = {};
161 size_t size = 0;
162 std::vector<bool> closed;
163 closed.reserve( contours.size() );
164 int numClosed = 0;
165 for ( const auto& c : contours )
166 {
167 if ( c.size() > 2 )
168 {
169 closed.push_back( c.front() == c.back() );
170 }
171 else
172 {
173 closed.push_back( false );
174 }
175 size += c.size();
176 if ( closed.back() )
177 ++numClosed;
178 }
179
180 reservePoints( size - numClosed );
181 vertResize( size - numClosed );
182
183 for ( int i = 0; i < contours.size(); ++i )
184 {
185 const auto& c = contours[i];
186 if ( c.empty() )
187 continue;
188 const auto e0 = makeEdge();
189 const auto v0 = addPoint( c[0] );
190 setOrg( e0, v0 );
191 auto e = e0;
192 for ( int j = 1; j + 1 < c.size(); ++j )
193 {
194 const auto ej = makeEdge();
195 splice( ej, e.sym() );
196 const auto vj = addPoint( c[j] );
197 setOrg( ej, vj );
198 e = ej;
199 }
200 if ( closed[i] )
201 {
202 splice( e0, e.sym() );
203 }
204 else
205 {
206 const auto vx = addPoint( c.back() );
207 setOrg( e.sym(), vx );
208 }
209 }
210 assert( isConsistentlyOriented() );
211}

213template<typename T, typename F>
214std::vector<std::vector<T>> PolylineTopology::convertToContours( F && getPoint ) const
215{
216 std::vector<std::vector<T>> res;
217
218 UndirectedEdgeBitSet linesUsed;
219 linesUsed.autoResizeSet( UndirectedEdgeId{ undirectedEdgeSize() } );
220 linesUsed.flip();
221 for ( EdgeId e0 : linesUsed )
222 {
223 if ( isLoneEdge( e0 ) )
224 continue;
225
226 EdgeId curLine = e0;
227 while ( curLine != next( curLine ) )
228 {
229 curLine = next( curLine ).sym();
230 if ( curLine == e0 )
231 break;
232 }
233
234 linesUsed.set( curLine.undirected(), false );
235
236 EdgeId e = curLine;
237 std::vector<T> cont;
238 cont.push_back( getPoint( org( e ) ) );
239 for ( ;; )
240 {
241 e = e.sym();
242 cont.push_back( getPoint( org( e ) ) );
243 e = next( e );
244 if ( !linesUsed.test_set( e.undirected(), false ) )
245 break;
246 }
247 res.push_back( std::move( cont ) );
248 }
249
250 return res;
251}
