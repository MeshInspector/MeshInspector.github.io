133template <typename T>
134Vector2<T> SymMatrix2<T>::eigens( Matrix2<T> * eigenvectors ) const
135{
136 //https://en.wikipedia.org/wiki/Eigenvalue_algorithm#2%C3%972_matrices
137 const auto tr = trace();
138 const auto q = tr / 2;
139 const auto p = std::sqrt( std::max( T(0), sqr( tr ) - 4 * det() ) ) / 2;
140 Vector2<T> eig;
141 if ( p <= std::abs( q ) * std::numeric_limits<T>::epsilon() )
142 {
143 // this is proportional to identity matrix
144 eig = { q, q };
145 if ( eigenvectors )
146 *eigenvectors = Matrix2<T>{};
147 return eig;
148 }
149 eig[0] = q - p;
150 eig[1] = q + p;
151 if ( eigenvectors )
152 {
153 const auto x = eigenvector( eig[0] ).normalized();
154 *eigenvectors = Matrix2<T>::fromRows( x, x.perpendicular() );
155 }
156 return eig;
157}

159template <typename T>
160Vector2<T> SymMatrix2<T>::eigenvector( T eigenvalue ) const
161{
162 const Vector2<T> row0( xx - eigenvalue, xy );
163 const Vector2<T> row1( xy, yy - eigenvalue );
164 // not-repeating eigenvalue means that one of two rows is not zero
165 const T lsq0 = row0.lengthSq();
166 const T lsq1 = row1.lengthSq();
167 return lsq0 >= lsq1 ? row0.perpendicular() : row1.perpendicular();
168}

170template <typename T>
171Vector2<T> SymMatrix2<T>::maxEigenvector() const
172{
173 const auto tr = trace();
174 const auto q = tr / 2;
175 const auto p = std::sqrt( std::max( T(0), sqr( tr ) - 4 * det() ) ) / 2;
176 if ( p <= std::abs( q ) * std::numeric_limits<T>::epsilon() )
177 {
178 // this is proportional to identity matrix
179 return Vector2<T>( T(1), T(0) );

184template <typename T>
185Vector2<T> SymMatrix2<T>::solve( const Vector2<T> & b, T tol ) const
186{
187 Matrix2<T> eigenvectors;
188 const auto eigenvalues = eigens( &eigenvectors );
189 const auto threshold = std::max( std::abs( eigenvalues[0] ), std::abs( eigenvalues[1] ) ) * tol;
190 Vector2<T> res;
191 for ( int i = 0; i < 2; ++i )
192 {
193 if ( std::abs( eigenvalues[i] ) <= threshold )
194 continue;
195 res += dot( b, eigenvectors[i] ) / eigenvalues[i] * eigenvectors[i];
196 }
197 return res;
198}
