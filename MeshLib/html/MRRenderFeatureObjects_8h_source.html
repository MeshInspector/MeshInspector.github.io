106// A common base class for sub-renderobjects that are combined into the proper features.
107// `ObjectType` is the underlying datamodel object that stores the mesh, e.g. `ObjectMesh`.
108// `RenderObjectType` is the underlying render object, e.g. `RenderMeshObject`.
109// If `IsPrimary` is true, the visual properties are copied from the target datamodel object.
110template <bool IsPrimary, typename ObjectType, typename RenderObjectType>
111using RenderFeatureComponent = RenderWrapObject::Wrapper<WrappedModelSubobject<IsPrimary, ObjectType>, RenderObjectType>;
