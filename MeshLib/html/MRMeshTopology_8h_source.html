433template <typename T>
434void MeshTopology::forEachVertex( const MeshTriPoint & p, T && callback ) const
435{
436 if ( auto v = p.inVertex( *this ) )
437 {
438 callback( v );
439 return;
440 }
441 if ( auto e = p.onEdge( *this ) )
442 {
443 callback( org( e.e ) );
444 callback( dest( e.e ) );
445 return;
446 }
447
448 VertId v[3];
449 getLeftTriVerts( p.e, v );
450 for ( int i = 0; i < 3; ++i )
451 callback( v[i] );
452}

454template<typename T>
455void MeshTopology::flipEdgesIn( const EdgeId e0, T && flipNeeded )
456{
457 EdgeId e = e0;
458 for (;;)
459 {
460 auto testEdge = prev( e.sym() );
461 if ( left( testEdge ) && right( testEdge ) && flipNeeded( testEdge ) )
462 flipEdge( testEdge );
463 else
464 {
465 e = next( e );
466 if ( e == e0 )
467 break; // full ring has been inspected
468 }
469 }
470}

472template<typename T>
473void MeshTopology::flipEdgesOut( EdgeId e0, T && flipNeeded )
474{
475 EdgeId e = e0;
476 for (;;)
477 {
478 if ( left( e ) && right( e ) && flipNeeded( e ) )
479 {
480 e0 = next( e );
481 flipEdge( e );
482 e = e0;
483 }
484 else
485 {
486 e = next( e );
487 if ( e == e0 )
488 break; // full ring has been inspected
489 }
490 }
491}

493// rearrange vector values by map (old.id -> new.id)
494template<typename T, typename I>
495[[nodiscard]] Vector<T, I> rearrangeVectorByMap( const Vector<T, I>& oldVector, const BMap<I, I>& map )
496{
497 Vector<T, I> newVector;
498 newVector.resize( map.tsize );
499
500 const auto& mData = map.b.data();
501 const auto sz = std::min( oldVector.size(), map.b.size() );
502 for ( I i = I(0); i < sz; ++i)
503 {
504 I newV = mData[i];
505 if ( newV.valid() )
506 newVector[newV] = oldVector[i];
507 }
508 return newVector;
509}
