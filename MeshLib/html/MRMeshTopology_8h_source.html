269 // data of every half-edge
270 struct HalfEdgeRecord
271 {
272 EdgeId next; // next counter clock wise half-edge in the origin ring
273 EdgeId prev; // next clock wise half-edge in the origin ring
274 VertId org; // vertex at the origin of the edge
275 FaceId left; // face at the left of the edge
276
277 bool operator ==( const HalfEdgeRecord & b ) const
278 { return next == b.next && prev == b.prev && org == b.org && left == b.left; }
279 bool operator !=( const HalfEdgeRecord & b ) const
