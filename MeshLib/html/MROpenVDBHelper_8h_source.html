156template <typename GredT>
157void translateToZero( GredT& grid)
158{
159 using TreeT = typename GredT::TreeType;
160 typename TreeT::Ptr outTreePtr = std::make_shared<TreeT>();
161 TreeT& inTree = grid.tree();
162 openvdb::tools::changeLevelSetBackground( *outTreePtr, inTree.background() );
163
164 openvdb::math::CoordBBox bbox = grid.evalActiveVoxelBoundingBox();
165 if ( bbox.empty() || bbox.min() == openvdb::math::Coord() )
166 return;
167
168 //using RangeProc = RangeProcessor<TreeT, translateValue<TreeT, bbox.min()>>;
169 using RangeProc = RangeProcessor<TreeT, ShiftTransformer<TreeT>>;
170 ShiftTransformer<TreeT> xform;
171 xform.setShift( -bbox.min() );
172 RangeProc proc( bbox, inTree, *outTreePtr, xform );
173
174 const auto gridClass = grid.getGridClass();
175 if ( gridClass != openvdb::GRID_LEVEL_SET )
176 {
177 // Independently transform the tiles of the input grid.
178 // Note: Tiles in level sets can only be background tiles, and they
179 // are handled more efficiently with a signed flood fill (see below).
180 typename RangeProc::TileIterT tileIter = inTree.cbeginValueAll();
181 tileIter.setMaxDepth( tileIter.getLeafDepth() - 1 ); // skip leaf nodes
182 typename RangeProc::TileRange tileRange( tileIter );
183 tbb::parallel_reduce( tileRange, proc );
184 }
185
186 typename RangeProc::LeafRange leafRange( inTree.cbeginLeaf() );
187 tbb::parallel_reduce( leafRange, proc );
188
189 if ( gridClass == openvdb::GRID_LEVEL_SET )
190 {
191 openvdb::tools::pruneLevelSet( *outTreePtr );
192 openvdb::tools::signedFloodFill( *outTreePtr );
193 }
194
195 grid.setTree( outTreePtr );
196}
