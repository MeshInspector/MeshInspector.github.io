160template <typename GredT>
161void translateToZero( GredT& grid)
162{
163 using TreeT = typename GredT::TreeType;
164 typename TreeT::Ptr outTreePtr = std::make_shared<TreeT>();
165 TreeT& inTree = grid.tree();
166 openvdb::tools::changeLevelSetBackground( *outTreePtr, inTree.background() );
167
168 openvdb::math::CoordBBox bbox = grid.evalActiveVoxelBoundingBox();
169 if ( bbox.empty() || bbox.min() == openvdb::math::Coord() )
170 return;
171
172 //using RangeProc = RangeProcessor<TreeT, translateValue<TreeT, bbox.min()>>;
173 using RangeProc = RangeProcessor<TreeT, ShiftTransformer<TreeT>>;
174 ShiftTransformer<TreeT> xform;
175 xform.setShift( -bbox.min() );
176 RangeProc proc( bbox, inTree, *outTreePtr, xform );
177
178 const auto gridClass = grid.getGridClass();
179 if ( gridClass != openvdb::GRID_LEVEL_SET )
180 {
181 // Independently transform the tiles of the input grid.
182 // Note: Tiles in level sets can only be background tiles, and they
183 // are handled more efficiently with a signed flood fill (see below).
184 typename RangeProc::TileIterT tileIter = inTree.cbeginValueAll();
185 tileIter.setMaxDepth( tileIter.getLeafDepth() - 1 ); // skip leaf nodes
186 typename RangeProc::TileRange tileRange( tileIter );
187 tbb::parallel_reduce( tileRange, proc );
188 }
189
190 typename RangeProc::LeafRange leafRange( inTree.cbeginLeaf() );
191 tbb::parallel_reduce( leafRange, proc );
192
193 if ( gridClass == openvdb::GRID_LEVEL_SET )
194 {
195 openvdb::tools::pruneLevelSet( *outTreePtr );
196 openvdb::tools::signedFloodFill( *outTreePtr );
197 }
198
199 grid.setTree( outTreePtr );
200}
