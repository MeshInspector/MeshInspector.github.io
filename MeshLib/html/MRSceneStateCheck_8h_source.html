72// check that given vector has exactly N objects if type ObjectT
73// returns error message if requirements are not satisfied
74template<typename ObjectT, bool visualRepresentationCheck>
75std::string sceneSelectedExactly( const std::vector<std::shared_ptr<const Object>>& objs, unsigned n )
76{
77 if ( objs.size() != n )
78 return "Exactly " + std::to_string( n ) + " " + ObjectT::TypeName() + "(s) must be selected";
79 for ( const auto& obj : objs )
80 {
81 auto tObj = obj->asType<ObjectT>();
82 if ( !tObj )
83 return std::string( "Selected object(s) must have type: " ) + ObjectT::TypeName();
84
85 if ( !ModelCheck::model( *tObj ) )
86 return "Selected object(s) must have valid model";
87
88 if constexpr ( visualRepresentationCheck )
89 if ( !hasVisualRepresentation( *tObj ) )
90 return "Selected object(s) must have valid visual representation";
91 }
92 return "";
93}

95// checks that given vector has at least N objects if type ObjectT
96// returns error message if requirements are not satisfied
97template<typename ObjectT, bool visualRepresentationCheck>
98std::string sceneSelectedAtLeast( const std::vector<std::shared_ptr<const Object>>& objs, unsigned n )
99{
100 if ( objs.size() < n )
101 return "At least " + std::to_string( n ) + " " + ObjectT::TypeName() + "(s) must be selected";
102 unsigned i = 0;
103 for ( const auto& obj : objs )
104 {
105 auto tObj = obj->asType<ObjectT>();
106 if ( !tObj )
107 continue;
108 if ( !ModelCheck::model( *tObj ) )
109 continue;
110 if constexpr ( visualRepresentationCheck )
111 if ( !hasVisualRepresentation( *tObj ) )
112 continue;
113 ++i;
114 }
115 return ( i >= n ) ?
116 "" :
117 ( "At least " + std::to_string( n ) + " selected object(s) must have type: " + ObjectT::TypeName() + " with valid model" );
118}

167template<typename ...Checks>
168class SceneStateOrCheck : virtual public Checks...
169{
170public:
171 virtual ~SceneStateOrCheck() = default;
172 virtual std::string isAvailable( const std::vector<std::shared_ptr<const Object>>&objs ) const override
173 {
174 std::vector<std::string> checkRes;
175 checkRes.reserve( sizeof...( Checks ) );
176 ( checkRes.push_back( Checks::isAvailable( objs ) ), ... );
177 std::string combinedRes;
178 for ( int i = 0; i + 1 < checkRes.size(); ++i )
179 {
180 if ( checkRes[i].empty() )
181 return "";
182 combinedRes += checkRes[i];
183 combinedRes += " or ";
184 }
185 if ( checkRes.back().empty() )
186 return "";
187 combinedRes += checkRes.back();
188 return combinedRes;
189 }
190};
