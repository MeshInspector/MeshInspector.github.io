15// check that given vector has exactly N objects if type ObjectT
16// returns error message if requirements are not satisfied
17template<typename ObjectT, bool visualRepresentationCheck, bool modelCheck>
18std::string sceneSelectedExactly( const std::vector<std::shared_ptr<const Object>>& objs, unsigned n )
19{
20 if ( objs.size() != n )
21 return "Exactly " + std::to_string( n ) + " " + ObjectT::TypeName() + "(s) must be selected";
22 for ( const auto& obj : objs )
23 {
24 auto tObj = dynamic_cast<const ObjectT*>( obj.get() );
25 if ( !tObj )
26 return std::string( "Selected object(s) must have type: " ) + ObjectT::TypeName();
27
28 if constexpr ( modelCheck )
29 if ( !tObj->hasModel() )
30 return "Selected object(s) must have valid model";
31
32 if constexpr ( visualRepresentationCheck )
33 if ( !tObj->hasVisualRepresentation() )
34 return "Selected object(s) must have valid visual representation";
35 }
36 return "";
37}

39// checks that given vector has at least N objects if type ObjectT
40// returns error message if requirements are not satisfied
41template<typename ObjectT, bool visualRepresentationCheck, bool modelCheck>
42std::string sceneSelectedAtLeast( const std::vector<std::shared_ptr<const Object>>& objs, unsigned n )
43{
44 if ( objs.size() < n )
45 return "At least " + std::to_string( n ) + " " + ObjectT::TypeName() + "(s) must be selected";
46 unsigned i = 0;
47 for ( const auto& obj : objs )
48 {
49 auto tObj = dynamic_cast<const ObjectT*>( obj.get() );
50 if ( !tObj )
51 continue;
52
53 if constexpr ( modelCheck )
54 if ( !tObj->hasModel() )
55 continue;
56
57 if constexpr ( visualRepresentationCheck )
58 if ( !tObj->hasVisualRepresentation() )
59 continue;
60 ++i;
61 }
62 return ( i >= n ) ?
63 "" :
64 ( "At least " + std::to_string( n ) + " selected object(s) must have type: " + ObjectT::TypeName() + " with valid model" );
65}
