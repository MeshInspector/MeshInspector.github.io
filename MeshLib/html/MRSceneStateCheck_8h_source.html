58// check that given vector has exactly N objects if type ObjectT
59// returns error message if requirements are not satisfied
60template<typename ObjectT>
61std::string sceneSelectedExactly( const std::vector<std::shared_ptr<const Object>>& objs, unsigned n )
62{
63 if ( objs.size() != n )
64 return "Exactly " + std::to_string( n ) + " " + ObjectT::TypeName() + "(s) must be selected";
65 for ( const auto& obj : objs )
66 {
67 auto tObj = obj->asType<ObjectT>();
68 if ( !tObj )
69 return std::string( "Selected object(s) must have type: " ) + ObjectT::TypeName();
70
71 if ( !ModelCheck::model( *tObj ) )
72 return "Selected object(s) must have valid model";
73 }
74 return "";
75}

77// checks that given vector has at least N objects if type ObjectT
78// returns error message if requirements are not satisfied
79template<typename ObjectT>
80std::string sceneSelectedAtLeast( const std::vector<std::shared_ptr<const Object>>& objs, unsigned n )
81{
82 if ( objs.size() < n )
83 return "At least " + std::to_string( n ) + " " + ObjectT::TypeName() + "(s) must be selected";
84 unsigned i = 0;
85 for ( const auto& obj : objs )
86 {
87 auto tObj = obj->asType<ObjectT>();
88 if ( !tObj )
89 continue;
90 if ( !ModelCheck::model( *tObj ) )
91 continue;
92 ++i;
93 }
94 return ( i >= n ) ?
95 "" :
96 ( "At least " + std::to_string( n ) + " selected object(s) must have type: " + ObjectT::TypeName() + " with valid model" );
97}
