52 T errorLimit = std::numeric_limits<T>::epsilon() * T( 20 ) )

56 return {};

72inline std::optional<Vector2f> intersection( const LineSegm2f& segm1, const LineSegm2f& segm2 )
73{
74 auto avec = segm1.b - segm1.a;
75 if ( cross( avec, segm2.a - segm1.a ) * cross( segm2.b - segm1.a, avec ) <= 0 )
76 return {};
77 auto bvec = segm2.b - segm2.a;
78 auto cda = cross( bvec, segm1.a - segm2.a );
79 auto cbd = cross( segm1.b - segm2.a, bvec );
80 if ( cda * cbd <= 0 )
81 return {};
82 return ( segm1.b * cda + segm1.a * cbd ) / ( cda + cbd );
83}

115template<typename T>
116LineSegm3<T> closestPoints( const Line3<T>& line1, const Line3<T>& line2 )
117{
118 const auto d11 = line1.d.lengthSq();
119 const auto d12 = dot( line1.d, line2.d );
120 const auto d22 = line2.d.lengthSq();
121 const auto det = d12 * d12 - d11 * d22;
122 if ( det == 0 )
123 {
124 // lines are parallel
125 return { line1.p, line2.project( line1.p ) };
126 }
127
128 const auto dp = line2.p - line1.p;
129 const auto x = dot( dp, line1.d ) / det;
130 const auto y = dot( dp, line2.d ) / det;
131 const auto a = d12 * y - d22 * x;
132 const auto b = d11 * y - d12 * x;
133 return { line1( a ), line2( b ) };
134}
