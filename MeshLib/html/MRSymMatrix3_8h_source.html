137{

140 return {};
141 SymMatrix3<T> res;
142 res.xx = ( yy * zz - yz * yz ) / det;
143 res.xy = ( xz * yz - xy * zz ) / det;
144 res.xz = ( xy * yz - xz * yy ) / det;
145 res.yy = ( xx * zz - xz * xz ) / det;
146 res.yz = ( xz * xy - xx * yz ) / det;
147 res.zz = ( xx * yy - xy * xy ) / det;

151template <typename T>
152Vector3<T> SymMatrix3<T>::eigens( Matrix3<T> * eigenvectors ) const
153{
154 //https://en.wikipedia.org/wiki/Eigenvalue_algorithm#3%C3%973_matrices
155 const auto q = trace() / 3;
156 const auto B = *this - diagonal( q );
157 const auto p2 = B.normSq();
158 const auto p = std::sqrt( p2 / 6 );
159 Vector3<T> eig;
160 if ( p <= std::abs( q ) * std::numeric_limits<T>::epsilon() )
161 {
162 // this is proportional to identity matrix
163 eig = { q, q, q };
164 if ( eigenvectors )
165 *eigenvectors = Matrix3<T>{};
166 return eig;
167 }
168 const auto r = B.det() / ( 2 * p * p * p );
169
170 // In exact arithmetic for a symmetric matrix - 1 <= r <= 1
171 // but computation error can leave it slightly outside this range.
172 if ( r <= -1 )
173 {
174 //phi = PI / 3;
175 eig[0] = q - 2 * p;
176 eig[1] = eig[2] = q + p;
177 if ( eigenvectors )
178 {
179 const auto x = eigenvector( eig[0] ).normalized();
180 const auto [ y, z ] = x.perpendicular();
181 *eigenvectors = Matrix3<T>::fromRows( x, y, z );
182 }
183 return eig;
184 }
185 if ( r >= 1 )
186 {
187 //phi = 0;
188 eig[0] = eig[1] = q - p;
189 eig[2] = q + 2 * p;
190 if ( eigenvectors )
191 {
192 const auto z = eigenvector( eig[2] ).normalized();
193 const auto [ x, y ] = z.perpendicular();
194 *eigenvectors = Matrix3<T>::fromRows( x, y, z );
195 }
196 return eig;
197 }
198 const auto phi = std::acos( r ) / 3;
199 eig[0] = q + 2 * p * cos( phi + T( 2 * PI / 3 ) );
200 eig[2] = q + 2 * p * cos( phi );
201 eig[1] = 3 * q - eig[0] - eig[2]; // 2 * q = trace() = eig[0] + eig[1] + eig[2]
202 if ( eigenvectors )
203 {
204 const auto x = eigenvector( eig[0] ).normalized();
205 const auto z = eigenvector( eig[2] ).normalized();
206 const auto y = cross( z, x );
207 *eigenvectors = Matrix3<T>::fromRows( x, y, z );
208 }
209 return eig;
210}

212template <typename T>
213Vector3<T> SymMatrix3<T>::eigenvector( T eigenvalue ) const
214{
215 const Vector3<T> row0( xx - eigenvalue, xy, xz );
216 const Vector3<T> row1( xy, yy - eigenvalue, yz );
217 const Vector3<T> row2( xz, yz, zz - eigenvalue );
218 // not-repeating eigenvalue means that some two rows are linearly independent
219 const Vector3<T> crs01 = cross( row0, row1 );
220 const Vector3<T> crs12 = cross( row1, row2 );
221 const Vector3<T> crs20 = cross( row2, row0 );
222 const T lsq01 = crs01.lengthSq();
223 const T lsq12 = crs12.lengthSq();
224 const T lsq20 = crs20.lengthSq();
225 if ( lsq01 > lsq12 )
226 {
227 if ( lsq01 > lsq20 )
228 return crs01;
229 }
230 else if ( lsq12 > lsq20 )
231 return crs12;
232 return crs20;
233}

235template <typename T>
236Vector3<T> SymMatrix3<T>::solve( const Vector3<T> & b, T tol, int * rank, Vector3<T> * space ) const
237{
238 Matrix3<T> eigenvectors;
239 const auto eigenvalues = eigens( &eigenvectors );
240 const auto threshold = std::max( std::abs( eigenvalues[0] ), std::abs( eigenvalues[2] ) ) * tol;
241 Vector3<T> res;
242 int myRank = 0;
243 for ( int i = 0; i < 3; ++i )
244 {
245 if ( std::abs( eigenvalues[i] ) <= threshold )
246 continue;
247 res += dot( b, eigenvectors[i] ) / eigenvalues[i] * eigenvectors[i];
248 ++myRank;
249 if ( space )
250 {
251 if ( myRank == 1 )
252 *space = eigenvectors[i];
253 else if ( myRank == 2 )
254 *space = cross( *space, eigenvectors[i] );
255 else
256 *space = Vector3<T>{};
257 }
258 }
259 if ( rank )
260 *rank = myRank;
261 return res;
262}
