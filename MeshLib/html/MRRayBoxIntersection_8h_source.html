63{
64 const Vector3i& sign = prec.sign;
65
66 // compare and update x-dimension with t0-t1
67 t1 = std::min( (box[sign.x].x - rayOrigin.p.x) * prec.invDir.x, t1 );
68 t0 = std::max( (box[1 - sign.x].x - rayOrigin.p.x) * prec.invDir.x, t0 );
