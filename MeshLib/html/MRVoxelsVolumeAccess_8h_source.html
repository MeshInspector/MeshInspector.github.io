157template <typename V>
158class VoxelsVolumeCachingAccessor
159{
160public:
161 using VolumeType = V;
162 using ValueType = typename V::ValueType;
163
164 struct Parameters
165 {
167 size_t preloadedLayerCount = 1;
168 };
169
170 VoxelsVolumeCachingAccessor( const VoxelsVolumeAccessor<V>& accessor, const VolumeIndexer& indexer, Parameters parameters = {} )
171 : accessor_( accessor )
172 , indexer_( indexer )
173 , params_( std::move( parameters ) )
174 , layers_( params_.preloadedLayerCount, std::vector<ValueType>( indexer_.sizeXY() ) )
175 {
176 assert( params_.preloadedLayerCount > 0 );
177 }
178
180 [[nodiscard]] int currentLayer() const
181 {
182 return z_;
183 }
184
186 void preloadLayer( int z )
187 {
188 assert( 0 <= z && z < indexer_.dims().z );
189 z_ = z;
190 for ( auto layerIndex = 0; layerIndex < layers_.size(); ++layerIndex )
191 {
192 if ( indexer_.dims().z <= z_ + layerIndex )
193 break;
194 preloadLayer_( layerIndex );
195 }
196 }
197
199 void preloadNextLayer()
200 {
201 z_ += 1;
202 for ( auto i = 0, j = 1; j < layers_.size(); ++i, ++j )
203 std::swap( layers_[i], layers_[j] );
204 if ( z_ + params_.preloadedLayerCount - 1 < indexer_.dims().z )
205 preloadLayer_( params_.preloadedLayerCount - 1 );
206 }
207
209 ValueType get( const Vector3i& pos ) const
210 {
211 const auto layerIndex = pos.z - z_;
212 if ( 0 <= layerIndex && layerIndex < layers_.size() )
213 return layers_[layerIndex][toLayerIndex( pos )];
214
215 return accessor_.get( pos );
216 }
217
218private:
219 [[nodiscard]] size_t toLayerIndex( const Vector3i& pos ) const
220 {
221 return indexer_.toVoxelId( { pos.x, pos.y, 0 } );
222 }
223
224 void preloadLayer_( size_t layerIndex )
225 {
226 assert( layerIndex < layers_.size() );
227 auto& layer = layers_[layerIndex];
228 const auto z = z_ + (int)layerIndex;
229 const auto& dims = indexer_.dims();
230 assert( 0 <= z && z < dims.z );
231 Vector3i pos { 0, 0, z };
232 for ( pos.y = 0; pos.y < dims.y; ++pos.y )
233 for ( pos.x = 0; pos.x < dims.x; ++pos.x )
234 layer[toLayerIndex( pos )] = accessor_.get( pos );
235 }
236
237private:
238 const VoxelsVolumeAccessor<V>& accessor_;
239 VolumeIndexer indexer_;
240 Parameters params_;
241
242 int z_ = -1;
243 std::vector<std::vector<ValueType>> layers_;
244};
