104 : accessor_( accessor )
105 , indexer_( indexer )
106 , params_( std::move( parameters ) )
107 , layers_( params_.preloadedLayerCount, std::vector<ValueType>( indexer_.sizeXY() ) )
108 {
109 assert( params_.preloadedLayerCount > 0 );
110 }

114 {

116 }
117
119 void preloadLayer( int z )
120 {
121 assert( 0 <= z && z < indexer_.dims().z );
122 z_ = z;
123 for ( auto layerIndex = 0; layerIndex < layers_.size(); ++layerIndex )
124 {
125 if ( indexer_.dims().z <= z_ + layerIndex )
126 break;
127 preloadLayer_( layerIndex );
128 }
129 }
130
132 void preloadNextLayer()
133 {
134 z_ += 1;
135 for ( auto i = 0, j = 1; j < layers_.size(); ++i, ++j )
136 std::swap( layers_[i], layers_[j] );
137 if ( z_ + params_.preloadedLayerCount - 1 < indexer_.dims().z )
138 preloadLayer_( params_.preloadedLayerCount - 1 );
139 }
140
142 ValueType get( const Vector3i& pos ) const
143 {
144 const auto layerIndex = pos.z - z_;
145 if ( 0 <= layerIndex && layerIndex < layers_.size() )
146 return layers_[layerIndex][toLayerIndex( pos )];
147
148 return accessor_.get( pos );
149 }
150
151private:
152 [[nodiscard]] size_t toLayerIndex( const Vector3i& pos ) const
153 {
154 return indexer_.toVoxelId( { pos.x, pos.y, 0 } );
155 }
156
157 void preloadLayer_( size_t layerIndex )
158 {
159 assert( layerIndex < layers_.size() );
160 auto& layer = layers_[layerIndex];
161 const auto z = z_ + (int)layerIndex;
162 const auto& dims = indexer_.dims();
163 assert( 0 <= z && z < dims.z );
164 Vector3i pos { 0, 0, z };
165 for ( pos.y = 0; pos.y < dims.y; ++pos.y )
166 for ( pos.x = 0; pos.x < dims.x; ++pos.x )
167 layer[toLayerIndex( pos )] = accessor_.get( pos );
168 }
