233template<typename T>
234struct RibbonMenuItemCall
235{
236 template<typename Func>
237 RibbonMenuItemCall( Func f )
238 {
239 static_assert( std::is_base_of_v<RibbonMenuItem, T> );
240 const auto& items = RibbonSchemaHolder::schema().items;
241 for ( const auto& item : items )
242 {
243 auto plugin = std::dynamic_pointer_cast<T>( item.second.item );
244 if ( !plugin )
245 continue;
246 f( plugin );
247 }
248 }
249};
