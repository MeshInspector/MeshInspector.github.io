272struct RibbonMenuItemAdder
273{
274 template<typename... Args>
275 RibbonMenuItemAdder(Args&&... args)
276 {
277 static_assert( std::is_base_of_v<RibbonMenuItem, T> );
278 RibbonSchemaHolder::addItem( std::make_shared<T>( std::forward<Args>(args)... ) );
279 }

282template<typename T>
283struct RibbonMenuItemCall
284{
285 template<typename Func>
286 RibbonMenuItemCall( Func f )
287 {
288 static_assert( std::is_base_of_v<RibbonMenuItem, T> );
289 const auto& items = RibbonSchemaHolder::schema().items;
290 for ( const auto& item : items )
291 {
292 auto plugin = std::dynamic_pointer_cast<T>( item.second.item );
293 if ( !plugin )
294 continue;
295 f( plugin );
296 }
297 }
298};
