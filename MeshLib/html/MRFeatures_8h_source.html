37 // * `dir` must be normalized.
38 // * Both `positiveLength` and `negativeLength` should be non-negative. They can be infinite (both or individually).

40 // * Both `positiveSideRadius` and `negativeSideRadius` must be non-negative.
41

234// Whether the distance can be computed between primitives.
235template <typename A, typename B>
236concept DistanceSupported = Traits::DistanceSupportedOneWay<A, B> || Traits::DistanceSupportedOneWay<B, A>;
237
238// Computes the distance between two primitives.
239template <typename A, typename B>
240requires DistanceSupported<A, B>
241[[nodiscard]] DistanceResult distance( const A& a, const B& b )
242{
243 if constexpr ( Traits::DistanceSupportedOneWay<A, B> )
244 {
245 DistanceResult ret = Traits::Binary<A, B>{}.distance( a, b );
246
247 if ( ret && ( !std::isfinite( ret.distance ) || !ret.closestPointA.isFinite() || !ret.closestPointB.isFinite() ) )
248 {
249 ret.status = DistanceResult::Status::not_finite;
250 return ret;
251 }
252
253 // Check that we got the correct distance here.
254 // Note that the distance is signed, so we apply `abs` to it to compare it properly.
255 assert( std::abs( ( ret.closestPointB - ret.closestPointA ).length() - std::abs( ret.distance ) ) < 0.0001f );
256
257 return ret;
258 }
259 else
260 {
261 static_assert( Traits::DistanceSupportedOneWay<B, A>, "This should never fail." );
262 DistanceResult ret = (distance)( b, a );
263 std::swap( ret.closestPointA, ret.closestPointB );
264 return ret;
265 }
266}
