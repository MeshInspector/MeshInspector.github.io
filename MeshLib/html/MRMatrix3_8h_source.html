132template <typename T>
133constexpr Matrix3<T> Matrix3<T>::rotation( const Vector3<T> & axis, T angle ) noexcept
134{
135 // https://en.wikipedia.org/wiki/Rotation_matrix#Rotation_matrix_from_axis_and_angle
136 auto u = axis.normalized();
137 T c = cos( angle );
138 T oc = 1 - c;
139 T s = sin( angle );
140 return {
141 { c + u.x * u.x * oc, u.x * u.y * oc - u.z * s, u.x * u.z * oc + u.y * s },
142 { u.y * u.x * oc + u.z * s, c + u.y * u.y * oc, u.y * u.z * oc - u.x * s },
143 { u.z * u.x * oc - u.y * s, u.z * u.y * oc + u.x * s, c + u.z * u.z * oc }
144 };
145}

147template <typename T>
148constexpr Matrix3<T> Matrix3<T>::rotation( const Vector3<T> & from, const Vector3<T> & to ) noexcept
149{
150 auto axis = cross( from, to );
151 if ( axis.lengthSq() > 0 )
152 return rotation( axis, angle( from, to ) );
153 if ( dot( from, to ) >= 0 )
154 return {}; // identity matrix
155 return rotation( cross( from, from.furthestBasisVector() ), T( PI ) );
156}

158template <typename T>
159constexpr Matrix3<T> Matrix3<T>::rotationFromEuler( const Vector3<T> & eulerAngles ) noexcept
160{
161 // https://www.geometrictools.com/Documentation/EulerAngles.pdf (36)
162 const auto cx = std::cos( eulerAngles.x );
163 const auto cy = std::cos( eulerAngles.y );
164 const auto cz = std::cos( eulerAngles.z );
165 const auto sx = std::sin( eulerAngles.x );
166 const auto sy = std::sin( eulerAngles.y );
167 const auto sz = std::sin( eulerAngles.z );
168 return {
169 { cy * cz, cz * sx * sy - cx * sz, cx * cz * sy + sx * sz },
170 { cy * sz, cx * cz + sx * sy * sz, -cz * sx + cx * sy * sz },
171 { -sy, cy * sx, cx * cy }
172 };
173}

184template <typename T>
185constexpr Matrix3<T> Matrix3<T>::inverse() const noexcept
186{
187 auto det = this->det();
188 if ( det == 0 )
189 return {};
190 return Matrix3<T>
191 {
192 { y.y * z.z - y.z * z.y, x.z * z.y - x.y * z.z, x.y * y.z - x.z * y.y },
193 { y.z * z.x - y.x * z.z, x.x * z.z - x.z * z.x, x.z * y.x - x.x * y.z },
194 { y.x * z.y - y.y * z.x, x.y * z.x - x.x * z.y, x.x * y.y - x.y * y.x }
195 } / det;
196}
