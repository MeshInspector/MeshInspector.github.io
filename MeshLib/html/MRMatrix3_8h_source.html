138template <typename T>
139constexpr Matrix3<T> Matrix3<T>::rotation( const Vector3<T> & axis, T angle ) noexcept
140{
141 // https://en.wikipedia.org/wiki/Rotation_matrix#Rotation_matrix_from_axis_and_angle
142 auto u = axis.normalized();
143 T c = cos( angle );
144 T oc = 1 - c;
145 T s = sin( angle );
146 return {
147 { c + u.x * u.x * oc, u.x * u.y * oc - u.z * s, u.x * u.z * oc + u.y * s },
148 { u.y * u.x * oc + u.z * s, c + u.y * u.y * oc, u.y * u.z * oc - u.x * s },
149 { u.z * u.x * oc - u.y * s, u.z * u.y * oc + u.x * s, c + u.z * u.z * oc }
150 };
151}

153template <typename T>
154constexpr Matrix3<T> Matrix3<T>::rotation( const Vector3<T> & from, const Vector3<T> & to ) noexcept
155{
156 auto axis = cross( from, to );
157 if ( axis.lengthSq() > 0 )
158 return rotation( axis, angle( from, to ) );
159 if ( dot( from, to ) >= 0 )
160 return {}; // identity matrix
161 return rotation( cross( from, from.furthestBasisVector() ), T( PI ) );
162}

164template <typename T>
165constexpr Matrix3<T> Matrix3<T>::rotationFromEuler( const Vector3<T> & eulerAngles ) noexcept
166{
167 // https://www.geometrictools.com/Documentation/EulerAngles.pdf (36)
168 const auto cx = std::cos( eulerAngles.x );
169 const auto cy = std::cos( eulerAngles.y );
170 const auto cz = std::cos( eulerAngles.z );
171 const auto sx = std::sin( eulerAngles.x );
172 const auto sy = std::sin( eulerAngles.y );
173 const auto sz = std::sin( eulerAngles.z );
174 return {
175 { cy * cz, cz * sx * sy - cx * sz, cx * cz * sy + sx * sz },
176 { cy * sz, cx * cz + sx * sy * sz, -cz * sx + cx * sy * sz },
177 { -sy, cy * sx, cx * cy }
178 };
179}
