28template <typename T>
29bool doTrianglesIntersect(
30 Vector3<T> a, Vector3<T> b, Vector3<T> c,
31 Vector3<T> d, Vector3<T> e, Vector3<T> f
32)
33{
34 const auto abcd = mixed( a - d, b - d, c - d );
35 const auto abce = mixed( a - e, b - e, c - e );
36 const auto abcf = mixed( a - f, b - f, c - f );
37 const auto abc_de = abcd * abce >= 0; // segment DE is located at one side of the plane ABC
38 const auto abc_fd = abcf * abcd >= 0; // segment FD is located at one side of the plane ABC
39
40 if ( abc_de && abc_fd && abce * abcf >= 0 )
41 return false; // triangle DEF is located at one side of the plane ABC
42
43 const auto defa = mixed( d - a, e - a, f - a );
44 const auto defb = mixed( d - b, e - b, f - b );
45 const auto defc = mixed( d - c, e - c, f - c );
46 const auto def_ab = defa * defb >= 0; // segment AB is located at one side of the plane DEF
47 const auto def_ca = defc * defa >= 0; // segment CA is located at one side of the plane DEF
48
49 if ( def_ab && def_ca && defb * defc >= 0 )
50 return false; // triangle ABC is located at one side of the plane DEF
51
52 if ( abc_de )
53 std::swap( d, f );
54 else if( abc_fd )
55 std::swap( d, e );
56 // now segments DE and FD are crossed by the plane ABC: D at one side and EF at the other
57
58 if ( def_ab )
59 std::swap( a, c );
60 else if ( def_ca )
61 std::swap( a, b );
62 // now segments AB and CA are crossed by the plane DEF: A at one side and BC at the other
63
64 const auto abde = mixed( a - e, b - e, d - e );
65 const auto abdf = mixed( a - f, b - f, d - f );
66
67 if ( abde * abdf < 0 )
68 return true; // AB segment penetrates triangle DEF since points E and F are at distinct sides of ABD
69
70 const auto acde = mixed( a - e, c - e, d - e );
71
72 if ( abde * acde < 0 )
73 return true; // DE segment penetrates triangle ABC since points B and C are at distinct sides of ADE
74
75 const auto acdf = mixed( a - f, c - f, d - f );
76
77 if ( acde * acdf < 0 )
78 return true; // AC segment penetrates triangle DEF since points E and F are at distinct sides of ACD
79
80 if ( abdf * acdf < 0 )
81 return true; // DF segment penetrates triangle ABC since points B and C are at distinct sides of ADF
82
83 return false;
84}

87template <typename T>
88bool doesEdgeXySeparate(
89 const Vector3<T> & x, const Vector3<T> & y, const Vector3<T> & z,
90 const Vector3<T> & u, const Vector3<T> & v, const Vector3<T> & w,
91 Vector3<T> d // approximate normal of the plane
92)
93{
94 const auto xy = ( y - x ).normalized();
95 d = ( d - xy * dot( xy, d ) ).normalized();
96 // now d is orthogonal to xy
97 const auto dz = dot( d, z - x );
98 return
99 dz * dot( d, u - x ) < 0 &&
100 dz * dot( d, v - x ) < 0 &&
101 dz * dot( d, w - x ) < 0;
102}

106template <typename T>
107bool doTrianglesIntersectExt(
108 const Vector3<T> & a, const Vector3<T> & b, const Vector3<T> & c,
109 const Vector3<T> & d, const Vector3<T> & e, const Vector3<T> & f
110)
111{
112 if ( !doTrianglesIntersect( a, b, c, d, e, f ) )
113 return false;
114
115 // direction from center to center
116 const auto dir = a + b + c - d - e - f;
117
118 return
119 !doesEdgeXySeparate( a, b, c, d, e, f, dir ) &&
120 !doesEdgeXySeparate( b, c, a, d, e, f, dir ) &&
121 !doesEdgeXySeparate( c, a, b, d, e, f, dir ) &&
122 !doesEdgeXySeparate( d, e, f, a, b, c, dir ) &&
123 !doesEdgeXySeparate( e, f, d, a, b, c, dir ) &&
124 !doesEdgeXySeparate( f, d, e, a, b, c, dir );
125}

129bool doTriangleLineIntersect(
130 const Vector3<T> & a, const Vector3<T> & b, const Vector3<T> & c,
131 const Vector3<T> & d, const Vector3<T> & e
132)
133{
134 const auto dabe = mixed( d - e, a - e, b - e );
135 const auto dbce = mixed( d - e, b - e, c - e );
136 if ( dabe * dbce <= 0 )
137 return false; // segment AC is located at one side of the plane DEB
138
139 const auto dcae = mixed( d - e, c - e, a - e );
140 if ( dbce * dcae <= 0 )
141 return false; // segment AB is located at one side of the plane DEC
142
143 if ( dcae * dabe <= 0 )
144 return false; // segment BC is located at one side of the plane DEA

164template <typename T>
165Vector3<T> findTriangleSegmentIntersection(
166 const Vector3<T>& a, const Vector3<T>& b, const Vector3<T>& c,
167 const Vector3<T>& d, const Vector3<T>& e
168)
169{
170 const auto abcd = std::abs( mixed( a - d, b - d, c - d ) );
171 const auto abce = std::abs( mixed( a - e, b - e, c - e ) );
172 auto r = std::clamp( abcd / ( abcd + abce ), T( 0 ), T( 1 ) );
173 return r * e + ( 1 - r ) * d;
174}

176template <typename T>
177std::optional<TriIntersectResult> rayTriangleIntersect_( const Vector3<T>& oriA, const Vector3<T>& oriB, const Vector3<T>& oriC,
178 const IntersectionPrecomputes<T>& prec )
179{
180 const T& Sx = prec.Sx;
181 const T& Sy = prec.Sy;
182 const T& Sz = prec.Sz;
183
184 const T Ax = oriA[prec.idxX] - Sx * oriA[prec.maxDimIdxZ];
185 const T Ay = oriA[prec.idxY] - Sy * oriA[prec.maxDimIdxZ];
186 const T Bx = oriB[prec.idxX] - Sx * oriB[prec.maxDimIdxZ];
187 const T By = oriB[prec.idxY] - Sy * oriB[prec.maxDimIdxZ];
188 const T Cx = oriC[prec.idxX] - Sx * oriC[prec.maxDimIdxZ];
189 const T Cy = oriC[prec.idxY] - Sy * oriC[prec.maxDimIdxZ];
190
191 // due to fused multiply-add (FMA): (A*B-A*B) can be different from zero, so we need epsilon
192 const T eps = std::numeric_limits<T>::epsilon() * std::max( { Ax, Bx, Cx, Ay, By, Cy } );
193 T U = Cx * By - Cy * Bx;
194 T V = Ax * Cy - Ay * Cx;
195 T W = Bx * Ay - By * Ax;
196
197 if( U < -eps || V < -eps || W < -eps )
198 {
199 if( U > eps || V > eps || W > eps )
200 {
201 // U,V,W have clearly different signs, so the ray misses the triangle
202 return std::nullopt;
203 }
204 }
205
206 T det = U + V + W;
207 if( det == T( 0 ) )
208 return std::nullopt;
209 const T Az = Sz * oriA[prec.maxDimIdxZ];
210 const T Bz = Sz * oriB[prec.maxDimIdxZ];
211 const T Cz = Sz * oriC[prec.maxDimIdxZ];
212 const T t = U * Az + V * Bz + W * Cz;
213
214 auto invDet = T( 1 ) / det;
215 return TriIntersectResult( float( V * invDet ), float( W * invDet ), float( t * invDet ) );
216}
