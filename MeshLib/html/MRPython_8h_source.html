230template<typename R, typename E, typename... Args>
231auto decorateExpected( std::function<Expected<R, E>( Args... )>&& f ) -> std::function<R( Args... )>
232{
233 return[fLocal = std::move( f )]( Args&&... args ) mutable -> R
234 {
235 auto res = fLocal(std::forward<Args>( args )...);
236 if (!res.has_value())
237 throwExceptionFromExpected(res.error());
238
239 if constexpr (std::is_void<R>::value)
240 return;
241 else
242 return res.value();
243 };
244}
