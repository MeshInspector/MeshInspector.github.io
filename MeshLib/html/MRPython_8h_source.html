150// overload `toString` functoion to throw exception from custom `Expected::error` type
151template<typename E>
152void throwExceptionFromExpected(const E& err)
153{
154 if constexpr (std::is_nothrow_convertible<E, std::string>::value)
155 throw std::runtime_error(err);
156 else
157 throw std::runtime_error(toString(err));
158}

160template<typename R, typename E, typename... Args>
161auto decorateExpected( std::function<Expected<R, E>( Args... )>&& f ) -> std::function<R( Args... )>
162{
163 return[fLocal = std::move( f )]( Args&&... args ) mutable -> R
164 {
165 auto res = fLocal(std::forward<Args>( args )...);
166 if (!res.has_value())
167 throwExceptionFromExpected(res.error());
168
169 if constexpr (std::is_void<R>::value)
170 return;
171 else
172 return res.value();
173 };
174}
175
176template<typename F>
177auto decorateExpected( F&& f )
178{
179 return decorateExpected( std::function( std::forward<F>( f ) ) );
180}
