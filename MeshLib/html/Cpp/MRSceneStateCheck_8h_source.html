19template<typename ObjectT>
20std::string getNObjectsLine( unsigned n )
21{
22 switch ( n )
23 {
24 case 1:
25 return std::string( "one " ) + ObjectT::StaticClassName();
26 case 2:
27 return std::string( "two " ) + ObjectT::StaticClassNameInPlural();
28 case 3:
29 return std::string( "three " ) + ObjectT::StaticClassNameInPlural();
30 case 4:
31 return std::string( "four " ) + ObjectT::StaticClassNameInPlural();
32 default:
33 return std::to_string( n ) + " " + ObjectT::StaticClassNameInPlural();
34 }
35}

39template<typename ObjectT, bool visualRepresentationCheck, bool modelCheck>
40std::string sceneSelectedExactly( const std::vector<std::shared_ptr<const Object>>& objs, unsigned n )
41{
42 if ( objs.size() != n )
43 return "Select exactly " + getNObjectsLine<ObjectT>( n );
44 for ( const auto& obj : objs )
45 {
46 auto tObj = dynamic_cast<const ObjectT*>( obj.get() );
47 if ( !tObj )
48 return std::string( "Selected object(s) must be " ) + ObjectT::StaticClassName();
49
50 if constexpr ( modelCheck )
51 if ( !tObj->hasModel() )
52 return "Selected object(s) must have valid model";
53
54 if constexpr ( visualRepresentationCheck )
55 if ( !tObj->hasVisualRepresentation() )
56 return "Selected object(s) must have valid visual representation";
57 }
58 return "";
59}

64std::string sceneSelectedAtLeast( const std::vector<std::shared_ptr<const Object>>& objs, unsigned n )
65{

233 virtual std::string isAvailable( const std::vector<std::shared_ptr<const Object>>&objs ) const override
234 {
235 std::vector<std::string> checkRes;
236 checkRes.reserve( sizeof...( Checks ) );
237 ( checkRes.push_back( Checks::isAvailable( objs ) ), ... );
238 std::string combinedRes;
239 for ( int i = 0; i < checkRes.size(); ++i )
240 {
241 if ( checkRes[i].empty() )
242 return "";
243
244 if ( i != 0 )
245 checkRes[i].front() = ( char )tolower( checkRes[i].front() );
246
247 combinedRes += checkRes[i];
248 if ( i + 1 < checkRes.size() )
249 combinedRes += " or ";
250 }
251 return combinedRes;
252 }
