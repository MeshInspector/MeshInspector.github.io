261Vector3<T> unitVector3( T azimuth, T altitude )
262{
263 const auto zenithAngle = T( PI2 ) - altitude;
264 return
265 {
266 std::sin( zenithAngle ) * std::cos( azimuth ),
267 std::sin( zenithAngle ) * std::sin( azimuth ),
