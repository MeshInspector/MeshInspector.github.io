40[[nodiscard]] V closestPointOnLineSegm( const V& pt, const LineSegm<V> & l )
41{
42 auto ab = l.b - l.a;
43 auto dt = dot( pt - l.a, ab );
44 auto abLengthSq = ab.lengthSq();
45 if ( dt <= 0 )
46 return l.a;
47 if ( dt >= abLengthSq )
48 return l.b;
49 auto ratio = dt / abLengthSq;
50 return l.a * ( 1 - ratio ) + l.b * ratio;
51}
