63template<typename T>
64[[nodiscard]] T triangleAspectRatio( const Vector3<T> & a, const Vector3<T> & b, const Vector3<T> & c )
65{
66 const auto bc = ( c - b ).length();
67 const auto ca = ( a - c ).length();
68 const auto ab = ( b - a ).length();
69 auto halfPerimeter = ( bc + ca + ab ) / 2;
70 auto den = 8 * ( halfPerimeter - bc ) * ( halfPerimeter - ca ) * ( halfPerimeter - ab );
71 if ( den <= 0 )
72 return std::numeric_limits<T>::max();
73
74 return bc * ca * ab / den;
75}

128template <typename T>
129[[nodiscard]] Triangle3<T> triangleWithNormal( const Triangle3<T> & t, const Vector3<T> & n )
130{
131 const auto c = ( t[0] + t[1] + t[2] ) / T(3);
132 Triangle3<T> res;
133 for ( int i = 0; i < 3; ++i )
134 {
135 const auto x = t[i] - c;
136 res[i] = x - n * dot( n, x );
137 }
138 if ( dot( n, cross( res[0], res[1] ) ) < 0 )
139 {
140 // projected triangle has inversed normal
141 int longest = 0;
142 T longestSq = 0;
143 for ( int i = 0; i < 3; ++i )
144 {
145 const auto sq = res[i].lengthSq();
146 if ( longestSq >= sq )
147 continue;
148 longest = i;
149 longestSq = sq;
150 }
151 const auto d = res[longest].normalized();
152 // project triangle the line (0, d)
153 for ( int i = 0; i < 3; ++i )
154 res[i] = d * dot( d, res[i] );
155 }
156
157 for ( int i = 0; i < 3; ++i )
158 res[i] += c;
159 return res;
160}

202template <typename T>
203[[nodiscard]] std::optional<Vector2<T>> posFromTriEdgeLengths( T a, T b, T c )
204{
205 if ( c == 0 )
206 {
207 if ( a == b )
208 return Vector2<T>{ a, 0 };
209 else
210 return {};
211 }
212 const auto aa = sqr( a );
213 const auto y = ( aa - sqr( b ) + sqr( c ) ) / ( 2 * c );
214 const auto yy = sqr( y );
215 if ( yy > aa )
216 return {};
217 const auto x = std::sqrt( aa - yy );
218 return Vector2<T>{ x, y };
219}

223template <typename T>
224[[nodiscard]] std::optional<T> quadrangleOtherDiagonal( T a, T b, T c, T a1, T b1 )
225{
226 const auto p = posFromTriEdgeLengths( a, b, c );
227 if ( !p )
228 return {};
229 auto p1 = posFromTriEdgeLengths( a1, b1, c );
230 if ( !p1 )
231 return {};
232 p1->x = -p1->x;
233 //where the other diagonal crosses axis Oy
234 auto y = ( p->x * p1->y - p1->x * p->y ) / ( p->x - p1->x );
235 if ( y < 0 || y > c )
236 return {};
237 return ( *p - *p1 ).length();
238}
