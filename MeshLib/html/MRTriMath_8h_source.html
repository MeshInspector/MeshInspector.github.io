16[[nodiscard]] T circumcircleDiameterSq( const Vector3<T> & a, const Vector3<T> & b, const Vector3<T> & c )
17{
18 const auto ab = ( b - a ).lengthSq();
19 const auto ca = ( a - c ).lengthSq();
20 const auto bc = ( c - b ).lengthSq();
21 if ( ab <= 0 )
22 return ca;
23 if ( ca <= 0 )
24 return bc;
25 if ( bc <= 0 )
26 return ab;
27 const auto f = cross( b - a, c - a ).lengthSq();
28 if ( f <= 0 )
29 return std::numeric_limits<T>::infinity();
30 return ab * ca * bc / f;
31}

42template <typename T>
43[[nodiscard]] Vector3<T> circumcircleCenter( const Vector3<T> & a, const Vector3<T> & b )
44{
45 const auto xabSq = cross( a, b ).lengthSq();
46 const auto aa = a.lengthSq();
47 const auto bb = b.lengthSq();
48 if ( xabSq <= 0 )
49 {
50 if ( aa <= 0 )
51 return b / T(2);
52 // else b == 0 || a == b
53 return a / T(2);
54 }
55 const auto ab = dot( a, b );
56 return ( bb * ( aa - ab ) * a + aa * ( bb - ab ) * b ) / ( 2 * xabSq );
57}

91{

109template<typename T>
110[[nodiscard]] T triangleAspectRatio( const Vector3<T> & a, const Vector3<T> & b, const Vector3<T> & c )
111{
112 const auto bc = ( c - b ).length();
113 const auto ca = ( a - c ).length();
114 const auto ab = ( b - a ).length();
115 auto halfPerimeter = ( bc + ca + ab ) / 2;
116 auto den = 8 * ( halfPerimeter - bc ) * ( halfPerimeter - ca ) * ( halfPerimeter - ab );
117 if ( den <= 0 )
118 return std::numeric_limits<T>::max();
119
120 return bc * ca * ab / den;
121}

174template <typename T>
175[[nodiscard]] Triangle3<T> triangleWithNormal( const Triangle3<T> & t, const Vector3<T> & n )
176{
177 const auto c = ( t[0] + t[1] + t[2] ) / T(3);
178 Triangle3<T> res;
179 for ( int i = 0; i < 3; ++i )
180 {
181 const auto x = t[i] - c;
182 res[i] = x - n * dot( n, x );
183 }
184 if ( dot( n, cross( res[0], res[1] ) ) < 0 )
185 {
186 // projected triangle has inversed normal
187 int longest = 0;
188 T longestSq = 0;
189 for ( int i = 0; i < 3; ++i )
190 {
191 const auto sq = res[i].lengthSq();
192 if ( longestSq >= sq )
193 continue;
194 longest = i;
195 longestSq = sq;
196 }
197 const auto d = res[longest].normalized();
198 // project triangle the line (0, d)
199 for ( int i = 0; i < 3; ++i )
200 res[i] = d * dot( d, res[i] );
201 }
202
203 for ( int i = 0; i < 3; ++i )
204 res[i] += c;
205 return res;
206}

248template <typename T>
249[[nodiscard]] std::optional<Vector2<T>> posFromTriEdgeLengths( T a, T b, T c )
250{
251 if ( c == 0 )
252 {
253 if ( a == b )
254 return Vector2<T>{ a, 0 };
255 else
256 return {};
257 }
258 const auto aa = sqr( a );
259 const auto y = ( aa - sqr( b ) + sqr( c ) ) / ( 2 * c );
260 const auto yy = sqr( y );
261 if ( yy > aa )
262 return {};
263 const auto x = std::sqrt( aa - yy );
264 return Vector2<T>{ x, y };
265}

269template <typename T>
270[[nodiscard]] std::optional<T> quadrangleOtherDiagonal( T a, T b, T c, T a1, T b1 )
271{
272 const auto p = posFromTriEdgeLengths( a, b, c );
273 if ( !p )
274 return {};
275 auto p1 = posFromTriEdgeLengths( a1, b1, c );
276 if ( !p1 )
277 return {};
278 p1->x = -p1->x;
279 //where the other diagonal crosses axis Oy
280 auto y = ( p->x * p1->y - p1->x * p->y ) / ( p->x - p1->x );
281 if ( y < 0 || y > c )
282 return {};
283 return ( *p - *p1 ).length();
284}
