24template <typename T>
25static std::pair<Vector3<T>, TriPoint<T>> closestPointInTriangle( const Vector3<T>& p, const Vector3<T>& a, const Vector3<T>& b, const Vector3<T>& c )
26{
27 const Vector3<T> ab = b - a;
28 const Vector3<T> ac = c - a;
29 const Vector3<T> ap = p - a;
30
31 const T d1 = dot( ab, ap );
32 const T d2 = dot( ac, ap );
33 if ( d1 <= 0 && d2 <= 0 )
34 return { a, { 0, 0 } };
35
36 const Vector3<T> bp = p - b;
37 const T d3 = dot( ab, bp );
38 const T d4 = dot( ac, bp );
39 if ( d3 >= 0 && d4 <= d3 )
40 return { b, { 1, 0 } };
41
42 const Vector3<T> cp = p - c;
43 const T d5 = dot( ab, cp );
44 const T d6 = dot( ac, cp );
45 if ( d6 >= 0 && d5 <= d6 )
46 return { c, { 0, 1 } };
47
48 const T vc = d1 * d4 - d3 * d2;
49 if ( vc <= 0 && d1 >= 0 && d3 <= 0 )
50 {
51 const T v = d1 / ( d1 - d3 );
52 return { a + v * ab, { v, 0 } };
53 }
54
55 const T vb = d5 * d2 - d1 * d6;
56 if ( vb <= 0 && d2 >= 0 && d6 <= 0 )
57 {
58 const T v = d2 / ( d2 - d6 );
59 return { a + v * ac, { 0, v } };
60 }
61
62 const T va = d3 * d6 - d5 * d4;
63 if ( va <= 0 && ( d4 - d3 ) >= 0 && ( d5 - d6 ) >= 0 )
64 {
65 const T v = ( d4 - d3 ) / ( ( d4 - d3 ) + ( d5 - d6 ) );
66 return { b + v * ( c - b ), { 1 - v, v } };
67 }
68
69 const T denom = 1 / ( va + vb + vc );
70 const T v = vb * denom;
71 const T w = vc * denom;
72 return { a + v * ab + w * ac, { v, w } };
73}
