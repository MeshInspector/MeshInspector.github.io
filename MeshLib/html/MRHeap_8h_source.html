10// stores map from element id in [0,size) to T;
11// and provides two operations:
12// 1) change the value of any element;
13// 2) find the element with the largest value

76template <typename T, typename I, typename P>
77void Heap<T, I, P>::setLargerValue( I elemId, const T & newVal )
78{
79 int pos = id2PosInHeap_[ elemId ];
80 assert( heap_[pos].id == elemId );
81 assert( !( pred_( newVal, heap_[pos].val ) ) );
82 heap_[pos].val = newVal;
83 while ( pos > 0 )
84 {
85 int parentPos = ( pos - 1 ) / 2;
86 if ( !( less_( parentPos, pos ) ) )
87 break;
88 auto parentId = heap_[parentPos].id;
89 assert( id2PosInHeap_[parentId] == parentPos );
90 std::swap( heap_[parentPos], heap_[pos] );
91 std::swap( parentPos, pos );
92 id2PosInHeap_[parentId] = parentPos;
93 }
94 id2PosInHeap_[elemId] = pos;
95}

97template <typename T, typename I, typename P>

99{
100 int pos = id2PosInHeap_[ elemId ];

102 assert( !( pred_( heap_[pos].val, newVal ) ) );
103 heap_[pos].val = newVal;
104 for (;;)
105 {
106 int child1Pos = 2 * pos + 1;
107 if ( child1Pos >= heap_.size() )
108 break;
109 auto child1Id = heap_[child1Pos].id;
110 int child2Pos = 2 * pos + 2;
111 if ( child2Pos >= heap_.size() )
112 {
113 assert( id2PosInHeap_[child1Id] == child1Pos );
114 if ( !( less_( child1Pos, pos ) ) )
115 {
116 std::swap( heap_[child1Pos], heap_[pos] );
117 std::swap( child1Pos, pos );
118 id2PosInHeap_[child1Id] = child1Pos;
119 }
120 break;

123 if ( !( less_( child1Pos, pos ) ) && !( less_( child1Pos, child2Pos ) ) )

126 std::swap( child1Pos, pos );

148{
149 const auto & a = heap_[posA];
150 const auto & b = heap_[posB];
